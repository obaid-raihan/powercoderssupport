				line-height: 2.5;
				text-transform: uppercase;
				text-align: center /* For the last element we can ignore the semicolon ";" */
			}

			}

			pre {
				display: block;
				width: 60%;
				margin: 0 auto 30px	;
				padding: 20px;
				background-color: white;
				white-space: pre-wrap;
				tab-size: 2;
			}


			nav {
				white-space: normal;
			}

			quote {
				color: #aaa;
			}

			//When several tags have the same styles
			h1, h2 {
				margin: 0;
				font-weight: normal;
				color: #212121;
			}

			//Parent - child relationship
			pre code {
				font-family: monospace;
				font-size: 1.2em;
			}

			code p {
				color: darkgrey;
			}

			nav ul {
				font-weight: normal;
			}

			nav ul li {
				color: #7a7a7a;
			}

			//Parent - child relationship, change only the first child
			nav > ul {
				font-weight: bold;
			}

			nav > ul > li {
				color: #212121;
			}

			//Parent - child relationship, change only the last child
			code p:last-child {
				color: orange;
			}

			//Parent - child relationship, ignore the first two children
			code p + p + p {
				color: #212121;
			}


			//Change styles on "hover"
			nav li:hover {
				text-indent: 20px;
			}

			nav li ul li:hover  {
				text-indent: 5px;
			}
